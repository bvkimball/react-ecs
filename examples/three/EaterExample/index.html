Eater Experiment

import {
    Teleporter,
    Teleporting
} from '@react-ecs/boids';
import {
    Entity,
    rad,
    useAnimationFrame,
    useECS
} from '@react-ecs/core';
import { ThreeView } from '@react-ecs/three';
import {
    Box,
    OrbitControls,
    Sphere,
    useHelper
} from '@react-three/drei';
import { Canvas } from '@react-three/fiber';
import * as _ from 'lodash';
import React, { useRef } from 'react';
import {
    DirectionalLightHelper,
    DirectionalLightShadow
} from 'three';


import {
    Rewarding,
    Sensing,
    Thinking,
    Thrusting,
    Turning
} from './facets';
import { RewardSystem, SensingSystem, ThinkingSystem, ThrustingSystem, TurningSystem } from './systems';


const Eater = (props) => {
    return (
        <Entity>
            <Sensing />
            <Thinking />
            <Turning />
            <Thrusting />
            <Teleporting />
            <ThreeView>
                <Box position={props.position} scale={[1, 1, 2]} castShadow receiveShadow>
                    <meshPhongMaterial color="green" />
                </Box>
            </ThreeView>


        </Entity>
    )
}


const Scene = () => {
    const lightRef = useRef()
    useHelper(lightRef, DirectionalLightHelper, 'royalblue', 1)
    useHelper(lightRef, DirectionalLightShadow, )
    return (
        <directionalLight
            ref={lightRef}
            position={[10, 10, 10]}
            castShadow
            shadow-mapSize-width={2048}
            shadow-mapSize-height={2048}
            shadow-camera-far={100}
            shadow-camera-left={-100}
            shadow-camera-right={100}
            shadow-camera-top={100}
            shadow-camera-bottom={-100}
        />
    );
}


export const EaterScene = () => {
    const ECS = useECS()
    useAnimationFrame(ECS.update)


    return (
        <Canvas shadows>
            <Scene />
            <OrbitControls
                minDistance={30}
                maxPolarAngle={rad(60)}
                minPolarAngle={rad(20)} />
            <Box scale={[200, 1, 200]} position={[0, -1, 0]} receiveShadow>
                <meshPhongMaterial color='lightblue' />
            </Box>
            <ECS.Provider>
                <SensingSystem />
                <ThinkingSystem />
                <TurningSystem speed={5} />
                <ThrustingSystem speed={10} />
                <RewardSystem />
                <Teleporter showBounds size={100} />


                {_.times(500, () => (
                    <Entity>
                        <Rewarding />
                        <ThreeView>
                            <Sphere position={[Math.random() * 200 - 100, 0, Math.random() * 200 - 100]}>
                                <meshPhongMaterial color='blue' />
                            </Sphere>
                        </ThreeView>
                    </Entity>
                ))}


                {_.times(100, () => (
                    <Eater position={[Math.random() * 200 - 100, 0, Math.random() * 200 - 100]} />
                ))}
            </ECS.Provider>
        </Canvas>
    )
}
