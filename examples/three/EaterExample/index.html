Eater Experiment

import { Organism } from '@alenaksu/neatjs';
import Network from '@alenaksu/neatjs/dist-types/network/Network';
import {
    Entity,
    Facet,
    rad,
    useAnimationFrame,
    useECS,
    useQuery,
    useSystem
} from '@react-ecs/core';
import { ThreeView } from '@react-ecs/three';
import {
    Box,
    OrbitControls,
    Sphere,
    useHelper
} from '@react-three/drei';
import { Canvas } from '@react-three/fiber';
import * as _ from 'lodash';
import React, {
    useEffect,
    useRef
} from 'react';
import {
    Color,
    DirectionalLightHelper,
    DirectionalLightShadow,
    Mesh,
    MeshPhongMaterial,
    Vector3
} from 'three';


import { Teleporter } from '../../../../../libs/boids/src/facets';
import { Teleporting } from '../../../../../libs/boids/src/facets/Teleporting';
import { TeleportSystem } from '../../../../../libs/boids/src/systems';
import { EaterSim } from './sim';


class Turning extends Facet<Turning> {
    amount?= 0;
}


const TurningSystem = ({ speed }) => {
    const query = useQuery(e => e.hasAll(ThreeView, Turning));
    return useSystem((dt) => {
        query.loop([ThreeView, Turning], (e, [view, turning]) => {
            const amt = turning.amount;
            const rot = view.object3d.rotation;
            rot.y += amt * dt * speed;
        })
    })
}


class Thrusting extends Facet<Thrusting> {
    amount?= 0;
}


const ThrustingSystem = ({ speed }) => {
    const query = useQuery(e => e.hasAll(ThreeView, Turning, Thrusting));
    return useSystem((dt) => {
        query.loop([ThreeView, Turning, Thrusting], (e, [view, turn, thrust]) => {
            view.object3d.translateZ(thrust.amount * dt * speed);
        });
    })
}


class Sensing extends Facet<Sensing> {
    direction?: Vector3
}


const SensingSystem = () => {
    const rewardQuery = useQuery(e => e.hasAll(ThreeView, Rewarding));
    const eaterQuery = useQuery(e => e.hasAll(ThreeView, Thinking));


    return useSystem(dt => {
        eaterQuery.loop([ThreeView, Sensing], (_, [eaterView, sense]) => {
            let closest: { dist: number, view: ThreeView } = undefined;
            rewardQuery.loop([ThreeView, Rewarding], (e, [view, reward]) => {
                const dist = eaterView.object3d.position.distanceTo(view.object3d.position);
                if (closest === undefined || dist < closest.dist) {
                    closest = { dist, view: view as ThreeView }
                }
            })
            sense.direction =
                closest.view.object3d.position
                    .clone()
                    .sub(eaterView.object3d.position)
                    .normalize()
        })


    })
}


const Eater = (props) => {
    const turning = new Vector3(
        0, Math.random() * 2 - 1, 0
    ).normalize()
    return (
        <Entity>
            <Sensing />
            <Thinking />
            <Turning />
            <Thrusting />
            <Teleporting />
            <ThreeView>
                <Box position={props.position} scale={[1, 1, 2]} castShadow receiveShadow>
                    <meshPhongMaterial color="green" />
                </Box>
            </ThreeView>


        </Entity>
    )
}


const Scene = () => {
    const lightRef = useRef()
    useHelper(lightRef, DirectionalLightHelper, 'royalblue', 1)
    useHelper(lightRef, DirectionalLightShadow, )
    return (
        <directionalLight
            ref={lightRef}
            position={[10, 10, 10]}
            castShadow
            shadow-mapSize-width={2048}
            shadow-mapSize-height={2048}
            shadow-camera-far={100}
            shadow-camera-left={-100}
            shadow-camera-right={100}
            shadow-camera-top={100}
            shadow-camera-bottom={-100}
        />
    );
}


class Thinking extends Facet<Thinking> {
    organism?: Organism;
    network?: Network;
}


const ThinkingSystem = () => {
    const sim = useRef<EaterSim>()
    const tick = useRef(0)
    const generation = useRef(0)


    const query = useQuery(e => e.hasAll(Thinking, ThreeView, Turning, Thrusting));


    const reset = () => {
        console.log("Resetting to tick 0.")
        tick.current = 0;


        if (generation.current > 0) {
            console.log(`New epoch: ${generation.current}.`)
            const winner = sim.current.population.getSuperChamp();
            console.log(`Fitness: ${winner.originalFitness}`);
            sim.current.population.epoch()
            console.log(`Population: ${sim.current.population.organisms.length}`)
        }


        const champ = sim.current.population.getSuperChamp();
        champ.getNetwork().neurons.values();


        query.loop([Thinking, ThreeView, Turning, Thrusting], (e, [brain, view, turn, thrust], i) => {
            ((view.object3d as Mesh).material as MeshPhongMaterial).color = new Color('green');
            brain.organism = sim.current.population.organisms[i];
            if (brain.organism) {
                brain.network = brain.organism.getNetwork();
                view.object3d.scale.set(1, 1, 2)
                view.object3d.position.set(
                    Math.random() * 200 - 100,
                    0,
                    Math.random() * 200 - 100,
                )
            } else {
                brain.network = undefined;
                turn.amount = 0;
                thrust.amount = 0;
                view.object3d.position.set(0, 10000, 0);
            }
        })


        generation.current++;
    }


    useEffect(() => {
        sim.current = new EaterSim();
        reset();
    }, [])


    return useSystem((dt) => {
        if (tick.current > 500) {
            reset()
        }


        query.loop([Thinking, Sensing, Turning, Thrusting], (e, [brain, sense, turn, thrust]) => {
            if (!brain.network) {
                turn.amount = 0;
                thrust.amount = 0;
                return;
            }
            const [ turnPower, thrustPower ] = brain.network.activate([
                1, sense.direction.x, sense.direction.y
            ])
            // console.log(`turn: ${turnPower.value}, thrust: ${thrustPower.value}`)
            turn.amount = turnPower.value * 2 - 1;
            thrust.amount = thrustPower.value * 2 - 1;
        })


        tick.current++;
    });
}


class Rewarding extends Facet<Rewarding> { }


const RewardSystem = () => {
    const rewardQuery = useQuery(e => e.hasAll(ThreeView, Rewarding));
    const eaterQuery = useQuery(e => e.hasAll(ThreeView, Thinking));


    return useSystem(dt => {
        eaterQuery.loop([ThreeView, Thinking], (_, [eaterView, brain]) => {
            rewardQuery.loop([ThreeView, Rewarding], (e, [rewardView, reward]) => {
                if (eaterView.object3d.position.distanceTo(rewardView.object3d.position) < 1) {
                    brain.organism.fitness ++;
                    ((eaterView.object3d as Mesh).material as MeshPhongMaterial).color.b += 10
                    eaterView.object3d.scale.z += 1;
                    rewardView.object3d.position.set(
                        Math.random() * 200 - 100,
                        0,
                        Math.random() * 200 - 100
                    );
                }
            })
        })
    })
}




export const EaterScene = () => {
    const ECS = useECS()
    useAnimationFrame(ECS.update)




    return (
        <Canvas shadows>
            <Scene />
            <OrbitControls
                minDistance={30}
                maxPolarAngle={rad(60)}
                minPolarAngle={rad(20)} />
            <Box scale={[200, 1, 200]} position={[0, -1, 0]} receiveShadow>
                <meshPhongMaterial color='lightblue' />
            </Box>
            <ECS.Provider>
                <SensingSystem />
                <ThinkingSystem />
                <TurningSystem speed={1} />
                <ThrustingSystem speed={5} />
                <RewardSystem />
                <Teleporter showBounds size={100} />


                {_.times(500, i => (
                    <Entity>
                        <Rewarding />
                        <ThreeView>
                            <Sphere position={[Math.random() * 200 - 100, 0, Math.random() * 200 - 100]}>
                                <meshPhongMaterial color='blue' />
                            </Sphere>
                        </ThreeView>
                    </Entity>
                ))}


                {_.times(100, i => (
                    <Eater position={[Math.random() * 200 - 100, 0, Math.random() * 200 - 100]} />
                ))}
            </ECS.Provider>
        </Canvas>
    )
}
